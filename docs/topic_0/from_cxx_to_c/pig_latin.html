string TranslateWord(string word) {

  if (vp == -1) {
      return (word);
  } else if (vp == 0) {
      return (ConcatString(word, "way"));
  } else {
      string head = SubString(word, 0, vp - 1);
      string tail = SubString(word, vp, StringLength(word) - 1);
      return (ConcatString(tail, ConcatString(head, "ay")));

}
